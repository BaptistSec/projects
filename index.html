#define SyscallTableSize (sizeof(SyscallTable) / sizeof(SyscallEntry))

GtkWidget *RiskScoreLabel;
int TotalRiskScore = 0;

void ResetSyscallFrequency() {
    for (int i = 0; i < SyscallTableSize; i++) {
        SyscallTable[i].Frequency = 0;
    }
}

void ExplainRiskScore(FILE *Report) {
    fprintf(Report, "\nDetailed Risk Explanation:\n");
    for (int i = 0; i < SyscallTableSize; i++) {
        if (SyscallTable[i].Frequency > 0) {
            fprintf(Report, "- Syscall: %s (Detected %d times)\n", SyscallTable[i].SyscallName, SyscallTable[i].Frequency);
            fprintf(Report, "  Technique: %s\n", SyscallTable[i].MitreTechnique);
            fprintf(Report, "  Explanation: %s\n", SyscallTable[i].Explanation);
            fprintf(Report, "  Contributed Risk Score: %d\n\n", SyscallTable[i].RiskScore);
        }
    }
}

// Hashing function to identify known malware signatures
void HashFileContent(const char *Content, unsigned char *Hash) {
    MD5((unsigned char *)Content, strlen(Content), Hash);
}

void CheckFileSignature(const char *Filename) {
    char *Content = ReadFileContent(Filename);
    if (Content != NULL) {
        unsigned char Hash[MD5_DIGEST_LENGTH];
        HashFileContent(Content, Hash);

        // Check against a list of known malware hashes
        const unsigned char KnownMalwareHash[MD5_DIGEST_LENGTH] = {/* known malware hash bytes */};

        if (memcmp(Hash, KnownMalwareHash, MD5_DIGEST_LENGTH) == 0) {
            printf("Known malware detected by hash in file: %s\n", Filename);
            TotalRiskScore += 100;
        }
        free(Content);
    }
}

// Heuristic analysis for detecting suspicious patterns
void HeuristicAnalysis(const char *Code) {
    if (strstr(Code, "execve") && strstr(Code, "fork")) {
        printf("Suspicious behaviour detected: Execve followed by fork.\n");
        TotalRiskScore += 50;
    }
}

// Entropy-based detection to find compressed or encrypted sections
double CalculateEntropy(const char *Content) {
    int ByteFrequencies[256] = {0};
    double Entropy = 0.0;
    size_t Length = strlen(Content);

    for (size_t i = 0; i < Length; i++) {
        ByteFrequencies[(unsigned char)Content[i]]++;
    }

    for (int i = 0; i < 256; i++) {
        if (ByteFrequencies[i] > 0) {
            double Probability = (double)ByteFrequencies[i] / Length;
            Entropy -= Probability * log2(Probability);
        }
    }
    return Entropy;
}

void DetectPackedOrEncryptedSections(const char *Content) {
    double Entropy = CalculateEntropy(Content);
    if (Entropy > 7.5) {
        printf("High entropy detected: Potential packed or encrypted section.\n");
        TotalRiskScore += 40;
    }
}

// Improved scan function to detect shellcode or obfuscated malware patterns
void ScanForMaliciousPatterns(const char *Code) {
    regex_t Regex;
    int Ret;

    const char *NopSledPattern = "(\\x90){10,}";
    Ret = regcomp(&Regex, NopSledPattern, REG_EXTENDED);
    if (Ret) {
        fprintf(stderr, "Could not compile regex\n");
        return;
    }

    Ret = regexec(&Regex, Code, 0, NULL, 0);
    if (!Ret) {
        printf("Potential shellcode detected: NOP sled found.\n");
        TotalRiskScore += 30;
    }

    regfree(&Regex);
}

void AnalyseFile(const char *Filename, FILE *Report) {
    printf("Analysing file: %s\n", Filename);

    char *Code = ReadFileContent(Filename);
    if (Code != NULL) {
        ResetSyscallFrequency();
        int RiskScore = CalculateRiskScore(Code);
        ScanForMaliciousPatterns(Code);
        HeuristicAnalysis(Code);
        DetectPackedOrEncryptedSections(Code);
        CheckFileSignature(Filename);

        fprintf(Report, "\nAnalysing file: %s\n", Filename);
        fprintf(Report, "Total Risk Score: %d/100\n", RiskScore);

        ExplainRiskScore(Report);
        free(Code);
    }
}

// Main function to handle file analysis and risk report generation
void GenerateReportForMultipleFiles(GSList *Filenames) {
    time_t Now = time(NULL);
    struct tm *TimeInfo = localtime(&Now);
    char Timestamp[32];
    strftime(Timestamp, sizeof(Timestamp), "%H%M%d%m%Y", TimeInfo);

    char ReportFilename[256];
    snprintf(ReportFilename, sizeof(ReportFilename), "analysis_report_%s.txt", Timestamp);

    FILE *Report = fopen(ReportFilename, "w");
    if (!Report) {
        printf("Failed to create report file.\n");
        return;
    }

    for (GSList *Iterator = Filenames; Iterator != NULL; Iterator = Iterator->next) {
        AnalyseFile((char *)Iterator->data, Report);
    }

    fclose(Report);
    printf("Report generated: %s\n", ReportFilename);

    SendReportUsingSCP(ReportFilename);
}
